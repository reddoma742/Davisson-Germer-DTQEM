\title{DTQEM: A Phenomenological Model for Correlated Path-Information and Thermal Decoherence (v18.0-C)}

\author{Reddouane Berramdane}
\email{reddoma@gmail.com}
\affiliation{Independent Researcher, Oujda, Morocco}

\date{\today}

\begin{abstract}
We present a phenomenological model for quantum decoherence in systems simultaneously subject to path-information extraction and thermal dephasing. The coherence function $C = C_0 \exp(-a I_{\text{path}} - b\Delta T/T_{\text{ref}} - c I_{\text{path}}\Delta T/T_{\text{ref}})$ extends the independent-bath baseline (v17.0-C, $c=0$) by a bilinear cross-coupling term $c$, capturing the synergistic action of correlated environmental modes. Calibrated at $T_{\text{ref}} = 300$ K, the conservative ($c=0$) limit achieves $R^2 = 0.9679$ and LOOCV $R^2 = 0.9356$; the full model reaches $R^2 = 0.9982$ (LOOCV $= 0.9814$) on the same synthetic dataset. Re-fitted to the eight-point dataset of Hornberger et al. (2003), the functional form yields $R^2 = 0.936$, demonstrating compatibility with molecule-interferometry data. A second-order cumulant analysis suggests strong inter-bath correlations, and an exploratory numerical scan reveals a thermal crossover at $T^* \approx 3.82$ K, flagged as a preliminary observation requiring low-temperature experimental confirmation. All code is openly available. A companion paper (Paper II) will provide the full microscopic derivation.
\end{abstract}

\section{Introduction}
Quantum decoherence---the irreversible loss of phase coherence in quantum superpositions---limits every practical quantum technology \cite{Breuer2002, Weiss2012}. Two physically distinct channels often coexist: (i) which-path information leakage, reducing fringe visibility via the Englert--Greenberger--Yasin (EGY) relation \cite{Englert1996}; and (ii) thermal dephasing from stochastic phase kicks.

In the simplest treatment, these channels add independently:
\begin{equation}
C_{\text{v17}} = C_0 \exp\left(-a I_{\text{path}} - b\frac{\Delta T}{T_{\text{ref}}}\right),
\label{eq:v17}
\end{equation}
where $I_{\text{path}} \in [0, 1]$ is the EGY distinguishability. However, in molecular and solid-state platforms, the same phonon bath often mediates both channels, producing a cross term that Eq.~\eqref{eq:v17} cannot capture. We formulate a minimal extension---DTQEM v18.0-C---adding a bilinear coupling $c I_{\text{path}} \Delta T/T_{\text{ref}}$, calibrate it, and test it against independent data. This paper focuses on the phenomenological model; a companion paper (Paper II) will provide the full microscopic derivation from the Feynman--Vernon influence functional.

\section{Model Definition}
\subsection{Core equation}
\begin{equation}
C(I_{\text{path}}, T) = C_0 \exp\left(-a I_{\text{path}} - b \frac{\Delta T}{T_{\text{ref}}} - c I_{\text{path}} \frac{\Delta T}{T_{\text{ref}}}\right),
\label{eq:v18}
\end{equation}
with $\Delta T = T - T_{\text{ref}}$, $T_{\text{ref}} = 300$ K fixed, and four free parameters $\{C_0, a, b, c\}$. Setting $c=0$ recovers Eq.~\eqref{eq:v17}.

\subsection{Physical interpretation}
\begin{itemize}
    \item $a I_{\text{path}}$: EGY path-dephasing, consistent with Caldeira--Leggett weak coupling \cite{Weiss2012}.
    \item $b\Delta T/T_{\text{ref}}$: linear thermal dephasing from the high-$T$ Ohmic expansion $\coth(\hbar\omega/2k_B T) \approx 2k_B T/\hbar\omega$.
    \item $c I_{\text{path}}\Delta T/T_{\text{ref}}$: phenomenological cross-coupling for shared bath modes. Motivated by physical boundary conditions and a second-order cumulant argument (Appendix~\ref{app:cumulant}); treated as a fit parameter.
\end{itemize}

\subsection{Operational mapping of $I_{\text{path}}$}
In circuit-QED, the mean intracavity photon number $\bar{n}$ gives \cite{Blais2021}
\begin{equation}
I_{\text{path}}(\bar{n}) = \sqrt{1 - e^{-\kappa \bar{n}}}.
\label{eq:ipath}
\end{equation}
This is a testable proposal for future experiments; it is not used in the calibration below.

\section{Validation on Hornberger (2003) Data}
Hornberger et al. \cite{Hornberger2003} reported C$_{70}$ fullerene fringe visibility vs. collision rate. We extracted the eight published points, mapped collision rate to $I_{\text{path}}$ via an auxiliary calibration, and re-fitted all four parameters independently. The re-fitted model achieves $R^2 = 0.936$. 

The agreement demonstrates that Eq.~\eqref{eq:v18} is compatible with independent molecule-interferometry data. It does not prove universality of the calibrated $c$. Statistical power of this $N=8$ test is quantified in Appendix~\ref{app:aicc}.

\section{Exploratory Temperature Scan: $T^*$}
We define the effective thermal coefficient
\begin{equation}
b_{\text{eff}}(T) \equiv - \left. \frac{\partial \ln C}{\partial (\Delta T/T_{\text{ref}})} \right|_{I_{\text{path}}}
\label{eq:beff}
\end{equation}
and evaluate it by numerically integrating the spin-boson dephasing functional with the Ohmic-Drude spectral density. Two regimes emerge (Fig.~\ref{fig:beff}):
\begin{itemize}
    \item $T > T^*$: $b_{\text{eff}} \propto T$ (classical Ohmic limit).
    \item $T < T^*$: $b_{\text{eff}}$ saturates to $b_{\text{sat}} \approx 0.0103$, as $\coth(\hbar\omega_c/2k_B T) \rightarrow 1$.
\end{itemize}
The crossover temperature is
\begin{equation}
T^* = \frac{\hbar\omega_c}{2k_B} \approx 3.82 \text{ K} \quad (\omega_c = 10^{12} \text{ rad s}^{-1}).
\label{eq:tstar}
\end{equation}

\caption{Effective thermal coefficient $b_{\text{eff}}(T)$. Blue dashed: classical extrapolation $b_0 T/T_{\text{ref}}$. Dotted: saturation $b_{\text{sat}} \approx 0.0103$. Dash-dotted: crossover $T^* \approx 3.82$ K.}

\noindent\textit{Important caveat.} Extrapolation below $\sim 50$ K violates the high-$T$ Ohmic assumption underlying the $b\Delta T$ term. $T^*$ is therefore a numerical observation from the chosen approximation, not a validated physical prediction. Experimental verification in the low-temperature regime is required.

\section{Assumptions and Limitations}
\begin{enumerate}
    \item \textbf{Markovian bath:} Memory effects are outside the model's scope.
    \item \textbf{Pure dephasing dominance:} Longitudinal relaxation ($T_1$) is neglected.
    \item \textbf{Room-temperature calibration:} The linear $b\Delta T$ term degrades below $\sim 50$ K.
    \item \textbf{$c$ is empirical:} Not derived microscopically in this work; specific to the calibration dataset. The full microscopic derivation is reserved for Paper II.
    \item \textbf{Small Hornberger dataset ($N=8$):} $N \geq 36$ is required for reliable model selection (Appendix~\ref{app:aicc}).
    \item \textbf{$I_{\text{path}}$ mapping errors:} Propagate into all reported fits.
    \item \textbf{$T^*$ is exploratory:} The crossover temperature is a numerical observation under specific approximations, not a rigorous prediction.
\end{enumerate}

\section{Discussion}
\subsection{Contribution}
DTQEM v18.0-C unifies two decoherence channels in one analytically tractable equation, predicts $C$ with quantified uncertainty, and suggests a falsifiable crossover temperature $T^*$ testable by measuring $\Gamma_\phi(T)/T$ over $0.5 \text{ K} \leq T \leq 100 \text{ K}$.

\subsection{Relation to prior work}
The bilinear form of Eq.~\eqref{eq:v18} has not, to our knowledge, appeared in the open quantum systems literature. The closest antecedents---Hornberger \cite{Hornberger2003} and Caldeira--Leggett \cite{Weiss2012}---both implicitly assume $c=0$.

\section{Outlook}
\begin{itemize}
    \item \textbf{Microscopic derivation (Paper II):} The companion paper will derive the cross term via the Feynman--Vernon influence functional for two correlated baths. Preliminary analysis gives an exact form $\propto 2\rho\sqrt{ab}\sqrt{I_{\text{path}} \cdot \Delta T/T_{\text{ref}}}$, of which Eq.~\eqref{eq:v18} is the leading-order polynomial approximation satisfying physical boundary conditions.
    \item \textbf{Low-temperature experimental test:} Confirm or refute the existence of a crossover near $T^*$ using cold-atom or superconducting qubit platforms.
    \item \textbf{Larger validation datasets:} $N \geq 36$ configurations (fullerene beams or transmon qubits) to robustly identify $c$.
\end{itemize}

\section{Conclusions}
DTQEM v18.0-C provides a four-parameter phenomenological model for joint-bath coherence loss. Key results:
\begin{itemize}
    \item Conservative calibration: $R^2 = 0.9679$, LOOCV $R^2 = 0.9356$ ($c=0$ limit).
    \item Independent validation: $R^2 = 0.936$ on Hornberger (2003), $N=8$.
    \item Cumulant analysis suggests strong inter-bath correlation.
    \item Numerical crossover $T^* \approx 3.82$ K, flagged as exploratory and requiring experimental verification.
\end{itemize}
A companion paper (Paper II) deriving the exact cross-coupling from first principles is in preparation.

\section{Motivation for the Cross-Coupling Term}
\label{app:cumulant}
Consider a two-level system coupled to a path-measurement bath (operator $\hat{B}_1$) and a thermal bath ($\hat{B}_2$), with total bath Hamiltonian $\hat{H}_B = \hat{H}_{B_1} + \hat{H}_{B_2} + \hat{H}_{B_1 B_2}$. A second-order cumulant expansion yields
\begin{equation}
\Phi_{\text{total}} = \Phi_1 + \Phi_2 + 2\rho\sqrt{\Phi_1 \Phi_2},
\label{eq:cumulant}
\end{equation}
where $\rho \in [0, 1]$ is the normalized cross-correlation coefficient. With $\Phi_1 \propto I_{\text{path}} \equiv u$ and $\Phi_2 \propto \Delta T/T_{\text{ref}} \equiv v$, the exact microscopic cross term is proportional to $\sqrt{uv}$.

Expanding $\sqrt{uv}$ around a nominal operating point $(u_0, v_0)$ gives a bilinear term. Matching the coefficient of $uv$ yields $c = \rho\sqrt{ab}/(2\sqrt{u_0 v_0})$. Using the calibrated $a,b,c$ from Table~\ref{tab:params} with representative values $u_0 = v_0 = 0.5$ suggests a cross-correlation coefficient of order unity ($\rho \sim 1$), indicating strong inter-bath coupling. The estimate is sensitive to the choice of $(u_0, v_0)$; the precise numerical value is deferred to Paper II, where the full microscopic derivation will also provide the rigorous expression for $\rho$.
